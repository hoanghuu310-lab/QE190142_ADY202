# 1. Khởi tạo Git
git init

# 2. Thêm tất cả các file vào danh sách chờ
git add .

# 3. Lưu trạng thái (Commit)
git commit -m "Khoi tao du an"

# 4. Đổi nhánh chính thành main
git branch -M main

# 5. Kết nối với GitHub (THAY LINK CỦA BẠN VÀO DÒNG DƯỚI)
git remote add origin https://github.com/hoanghuu310-lab/QE190142_ADY202.git

# 6. Đẩy code lên
git push -u origin main
